-- 기본 예시
-- A.REGIONID = B.REGIONID인 B의 REGIONNAME만 조회된다.
SELECT A.*,  B.REGIONNAME
FROM KOPO_JOIN_JOINSELLOUT A
LEFT JOIN KOPO_JOIN_JOINMASTER B
ON A.REGIONID = B.REGIONID;

SELECT A.REGIONID, B.REGIONNAME, A.PRODUCT, A.YEARWEEK, A.QTY
FROM KOPO_JOIN_JOINSELLOUT A
LEFT JOIN KOPO_JOIN_JOINMASTER B
ON A.REGIONID = B.REGIONID;

-- 조인키 두 개 예시
SELECT A.REGIONID, A.PRODUCT, A.YEARWEEK, A.QTY,
       B.PROMOTION
FROM KOPO_JOIN_UNIONDATA A
LEFT JOIN KOPO_JOIN_PROMODATA B
ON A.REGIONID = B.REGION
AND A.PRODUCT = B.PD;
